            for(let i=6; i<=9; i++) {

                const rawAnswer = document.getElementById(`q${i}`).value;

                studentAnswers[`q${i}`] = rawAnswer; 

                const answer = cleanString(rawAnswer);

                const possibleAnswers = correctAnswers[`q${i}`];

                if (possibleAnswers.includes(answer)) {

                    rawScore++;

                }

            }


            // Calculation: 9 points from questions + 1 point default = Final Grade out of 10

            let finalGrade = rawScore + 1;


            showResults(finalGrade, studentAnswers);


            try {

                const studentDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'modal_test_attempts', activeStudentName);

                await setDoc(studentDocRef, {

                    name: activeStudentName,

                    score: finalGrade,

                    maxScore: 10,

                    answers: studentAnswers,

                    completedAt: new Date().toISOString()

                });

            } catch (error) {

                console.warn("Error saving to database:", error);

            }

        }


        function showResults(finalGrade, studentAnswers) {

            resultsModal.classList.remove('hidden-view');

            

            document.getElementById('resName').textContent = activeStudentName;

            document.getElementById('scoreDisplay').textContent = finalGrade;


            const summaryDiv = document.getElementById('answersSummary');

            let summaryHTML = '<h3 class="font-bold text-gray-800 mb-4 border-b pb-2">Summary: Your Answers vs. Correct Answers</h3>';

            summaryHTML += '<div class="space-y-3">';


            for(let i=1; i<=9; i++) {

                let ansText = studentAnswers[`q${i}`];

                let isCorrect = false;

                

                if (i <= 5) {

                    isCorrect = ansText === correctAnswers[`q${i}`];

                    if(ansText) {

                        ansText = document.querySelector(`#q${i} option[value="${ansText}"]`).text;

                    } else {

                        ansText = 'No answer provided';

                    }

                } else {

                    let cleanAns = cleanString(ansText);

                    isCorrect = correctAnswers[`q${i}`].includes(cleanAns);

                    if(!ansText || ansText.trim() === '') ansText = 'No answer provided';

                }


                let color = isCorrect ? 'text-green-600' : 'text-red-500';

                let icon = isCorrect ? '✓' : '✗';

                

                summaryHTML += `

                <div class="bg-gray-50 p-3 rounded-lg border border-gray-200 shadow-sm">

                    <p class="font-bold text-gray-700 text-xs uppercase mb-1">Question ${i}</p>

                    <p class="text-sm">Your answer: <span class="${color} font-bold">${ansText} ${icon}</span></p>

                    ${!isCorrect ? `<p class="text-indigo-600 font-semibold text-xs mt-1 pt-1 border-t border-gray-200">Correct answer: <strong>${displayCorrectAnswers[`q${i}`]}</strong></p>` : ``}

                </div>`;

            }

            

            summaryHTML += '</div>';

            summaryDiv.innerHTML = summaryHTML;

        }


        function generateJPG() {

            const manualBtn = document.getElementById('manualDownloadBtn');

            const originalText = manualBtn.innerHTML;

            manualBtn.innerHTML = 'Processing image...';

            manualBtn.disabled = true;


            html2canvas(document.getElementById('modalPrintArea'), {

                backgroundColor: "#ffffff",

                scale: 2, 

                useCORS: true 

            }).then(canvas => {

                const dataUrl = canvas.toDataURL('image/jpeg', 0.9);

                

                try {

                    const link = document.createElement('a');

                    link.download = `Grade_${activeStudentName.replace(/\s+/g, '_')}.jpg`;

                    link.href = dataUrl;

                    document.body.appendChild(link);

                    link.click();

                    document.body.removeChild(link);

                } catch(err) {

                    console.warn("Automatic download was blocked by browser.");

                }

                

                const fallbackContainer = document.getElementById('fallbackImageContainer');

                fallbackContainer.innerHTML = '<p class="text-sm text-green-700 font-bold mb-3 flex items-center"><svg class="w-5 h-5 mr-1" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M13 16h-1v-4h-1m1-4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg> If the file didn\'t download automatically, long-press (or right-click) the image below and select "Save image":</p>';

                

                const img = document.createElement('img');

                img.src = dataUrl;

                img.className = "w-full rounded-lg border-2 border-gray-300 shadow-md";

                fallbackContainer.appendChild(img);

                fallbackContainer.classList.remove('hidden-view');

                

                manualBtn.innerHTML = '<svg class="w-5 h-5 mr-2" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path></svg> Re-generate Image';

                manualBtn.disabled = false;

            }).catch(err => {

                console.error("Generation error:", err);

                manualBtn.innerHTML = originalText;

                manualBtn.disabled = false;

                const fallbackContainer = document.getElementById('fallbackImageContainer');

                fallbackContainer.innerHTML = '<p class="text-sm text-red-600 font-bold mb-3">Error: Could not generate image. Please take a manual screenshot of this window.</p>';

                fallbackContainer.classList.remove('hidden-view');

            });

        }


        document.getElementById('manualDownloadBtn').addEventListener('click', generateJPG);


        document.getElementById('closeModalBtn').addEventListener('click', () => {

            resultsModal.classList.add('hidden-view');

        });
